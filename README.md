int Leap(int iNumber1,int iNumber2)
{
	int leap,a,i;
	for(i=iNumber1;i<=iNumber2-iNumber1;i++)
	{
		if(i % 4 == 0)
		{if(i % 100 == 0)
		{if(i % 400 == 0)
				{leap=1;}
		else
				{leap=0;}
		}
		else{leap=1;}
		}
		else
			{leap=0;}
		if(leap==1)
		{a=a+1;}
	}
	return a;
}
